PongerAgent - Local Methods per Second

Local access of requests by another blade in the same reactor are quite fast. We will use PongerAgent, a sync agent of Ponger, to see just how fast local access can be. 

import org.agilewiki.jactor2.core.blades.SyncAgentBase;
import org.agilewiki.jactor2.core.messages.SyncRequest;

public class PongerAgent extends SyncAgentBase {
    private final long count;
    
    public static SyncRequest startSReq(
            final Ponger _ponger, 
            final long _count) throws Exception {
        return new PongerAgent(_ponger, _count).startSReq();
    }

    public PongerAgent(
            final Ponger _ponger, 
            final long _count) throws Exception {
        super(_ponger);
        count = _count;
    }
    
    @Override
    public Void start() throws Exception {
        long i = 0;
        while (i < count) {
            i++;
            local(localBlade.pingSReq());
        }
        return null;
    }
}
